This example demonstrates how to build an account management interface using chat:

- Profile updates
- Password changes with validation
- Email verification
- Account deletion with confirmation
- Security settings
- Privacy preferences

## Complete Example

```tsx
import { useMemo } from 'react';
import {
  Chat,
  createRequestInputButton,
  createRequestConfirmationButton,
} from 'actionable-support-chat';
import { useChatStore } from 'actionable-support-chat';
import type { Message } from 'actionable-support-chat';

function AccountManagementChat() {
  const { addMessage } = useChatStore();

  const initialMessages: readonly Message[] = useMemo(
    () => [
      {
        id: 1,
        type: 'other',
        content: '⚙️ **Account Settings**\n\nWhat would you like to manage?',
        timestamp: new Date(),
        buttons: [
          {
            label: 'Profile Settings',
            variant: 'info',
            onClick: () => handleProfileSettings(),
          },
          {
            label: 'Security',
            variant: 'warning',
            onClick: () => handleSecurity(),
          },
          {
            label: 'Privacy',
            variant: 'default',
            onClick: () => handlePrivacy(),
          },
          {
            label: 'Delete Account',
            variant: 'error',
            onClick: () => handleDeleteAccount(),
          },
        ],
      },
    ],
    []
  );

  const handleProfileSettings = () => {
    addMessage({
      type: 'other',
      content: '👤 **Profile Settings**\n\nWhat would you like to update?',
      buttons: [
        createRequestInputButton({
          initialLabel: 'Change Name',
          inputPromptMessage: 'Enter your new name:',
          inputType: 'text',
          placeholder: 'John Doe',
          inputDescription: 'This is how your name appears to others',
          validator: (value) => {
            if (value.trim().length < 2) {
              return 'Name must be at least 2 characters';
            }
            if (value.trim().length > 50) {
              return 'Name must be less than 50 characters';
            }
            return true;
          },
          onInput: async (name) => {
            addMessage({
              type: 'other',
              content: 'Updating your name...',
            });

            await new Promise(resolve => setTimeout(resolve, 1000));

            addMessage({
              type: 'other',
              content: `✅ Name updated to "${name}" successfully!`,
              buttons: [
                {
                  label: 'Back to Profile',
                  onClick: () => handleProfileSettings(),
                },
                {
                  label: 'Main Menu',
                  onClick: () => handleBackToMenu(),
                },
              ],
            });
          },
        }),
        createRequestInputButton({
          initialLabel: 'Update Email',
          inputPromptMessage: 'Enter your new email address:',
          inputType: 'email',
          placeholder: 'your.email@example.com',
          inputDescription: 'We will send a verification email',
          validator: (value) => {
            const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            return emailRegex.test(value) || 'Please enter a valid email address';
          },
          onInput: async (email) => {
            addMessage({
              type: 'other',
              content: 'Sending verification email...',
            });

            await new Promise(resolve => setTimeout(resolve, 1500));

            addMessage({
              type: 'other',
              content: `✅ Verification email sent to ${email}!\n\nPlease check your inbox and click the verification link to confirm your new email address.`,
              buttons: [
                {
                  label: 'Resend Email',
                  variant: 'info',
                  onClick: async () => {
                    addMessage({
                      type: 'other',
                      content: 'Resending verification email...',
                    });
                    await new Promise(resolve => setTimeout(resolve, 1000));
                    addMessage({
                      type: 'other',
                      content: '✅ Verification email resent!',
                    });
                  },
                },
                {
                  label: 'Main Menu',
                  onClick: () => handleBackToMenu(),
                },
              ],
            });
          },
        }),
        createRequestInputButton({
          initialLabel: 'Update Phone',
          inputPromptMessage: 'Enter your phone number:',
          inputType: 'tel',
          placeholder: '+1 (555) 123-4567',
          inputDescription: 'Used for two-factor authentication',
          validator: (value) => {
            const phoneRegex = /^\+?[\d\s\-\(\)]+$/;
            if (!phoneRegex.test(value)) {
              return 'Please enter a valid phone number';
            }
            if (value.replace(/\D/g, '').length < 10) {
              return 'Phone number must be at least 10 digits';
            }
            return true;
          },
          onInput: async (phone) => {
            addMessage({
              type: 'other',
              content: 'Updating phone number...',
            });

            await new Promise(resolve => setTimeout(resolve, 1000));

            addMessage({
              type: 'other',
              content: `✅ Phone number updated to ${phone}!`,
              buttons: [
                {
                  label: 'Main Menu',
                  onClick: () => handleBackToMenu(),
                },
              ],
            });
          },
        }),
        {
          label: 'Back to Menu',
          variant: 'dull',
          onClick: () => handleBackToMenu(),
        },
      ],
    });
  };

  const handleSecurity = () => {
    addMessage({
      type: 'other',
      content: '🔒 **Security Settings**\n\nManage your account security:',
      buttons: [
        createRequestInputButton({
          initialLabel: 'Change Password',
          inputPromptMessage: 'Enter your new password:',
          inputType: 'password',
          placeholder: 'Enter new password',
          inputDescription: 'Must be 8+ chars with uppercase, lowercase, and number',
          validator: (value) => {
            if (value.length < 8) {
              return 'Password must be at least 8 characters long';
            }
            if (!/[A-Z]/.test(value)) {
              return 'Password must contain at least one uppercase letter';
            }
            if (!/[a-z]/.test(value)) {
              return 'Password must contain at least one lowercase letter';
            }
            if (!/[0-9]/.test(value)) {
              return 'Password must contain at least one number';
            }
            if (!/[!@#$%^&*]/.test(value)) {
              return 'Password must contain at least one special character (!@#$%^&*)';
            }
            return true;
          },
          onInput: async (password) => {
            addMessage({
              type: 'other',
              content: 'Updating your password...',
            });

            await new Promise(resolve => setTimeout(resolve, 1000));

            addMessage({
              type: 'other',
              content: '✅ Password changed successfully!\n\nFor security, you\'ve been logged out of all other devices.',
              buttons: [
                {
                  label: 'Main Menu',
                  onClick: () => handleBackToMenu(),
                },
              ],
            });
          },
        }),
        {
          label: 'Enable 2FA',
          variant: 'success',
          onClick: () => handle2FA(),
        },
        {
          label: 'View Login History',
          variant: 'info',
          onClick: () => handleLoginHistory(),
        },
        createRequestConfirmationButton({
          initialLabel: 'Revoke All Sessions',
          confirmationMessage: 'This will log you out of all devices except this one. Continue?',
          variant: 'warning',
          onConfirm: async () => {
            addMessage({
              type: 'other',
              content: 'Revoking all sessions...',
            });

            await new Promise(resolve => setTimeout(resolve, 1000));

            addMessage({
              type: 'other',
              content: '✅ All sessions revoked! You\'re now logged out of all other devices.',
              buttons: [
                {
                  label: 'Main Menu',
                  onClick: () => handleBackToMenu(),
                },
              ],
            });
          },
          onReject: () => {
            addMessage({
              type: 'other',
              content: 'Sessions remain active.',
            });
          },
        }),
        {
          label: 'Back to Menu',
          variant: 'dull',
          onClick: () => handleBackToMenu(),
        },
      ],
    });
  };

  const handle2FA = () => {
    addMessage({
      type: 'other',
      content: '🔐 **Two-Factor Authentication**\n\n2FA adds an extra layer of security to your account. Enable it now?',
      buttons: [
        {
          label: 'Enable 2FA',
          variant: 'success',
          onClick: async () => {
            addMessage({
              type: 'other',
              content: 'Setting up 2FA...',
            });

            await new Promise(resolve => setTimeout(resolve, 1500));

            addMessage({
              type: 'other',
              content: '✅ 2FA enabled!\n\nScan this QR code with your authenticator app:\n\n[QR Code would be displayed here]\n\n**Backup codes:**\n- 1234-5678\n- 8765-4321\n- 9012-3456\n\nSave these codes in a safe place!',
              buttons: [
                {
                  label: 'I\'ve Saved the Codes',
                  variant: 'success',
                  onClick: () => {
                    addMessage({
                      type: 'other',
                      content: '✅ 2FA setup complete! Your account is now more secure.',
                    });
                  },
                },
              ],
            });
          },
        },
        {
          label: 'Not Now',
          variant: 'dull',
          onClick: () => {
            addMessage({
              type: 'other',
              content: 'You can enable 2FA anytime from Security settings.',
            });
          },
        },
      ],
    });
  };

  const handleLoginHistory = () => {
    addMessage({
      type: 'other',
      content: '📊 **Login History**\n\n**Recent logins:**\n\n1. **Current session**\n   • Chrome on Windows\n   • Just now\n   • Location: New York, US\n\n2. Chrome on Windows\n   • 2 hours ago\n   • Location: New York, US\n\n3. Safari on iPhone\n   • Yesterday at 3:45 PM\n   • Location: New York, US',
      buttons: [
        {
          label: 'Back to Security',
          onClick: () => handleSecurity(),
        },
        {
          label: 'Main Menu',
          onClick: () => handleBackToMenu(),
        },
      ],
    });
  };

  const handlePrivacy = () => {
    addMessage({
      type: 'other',
      content: '🔐 **Privacy Settings**\n\nManage your privacy preferences:',
      buttons: [
        {
          label: 'Profile Visibility',
          variant: 'info',
          onClick: () => {
            addMessage({
              type: 'other',
              content: '👁️ **Profile Visibility**\n\nWho can see your profile?',
              buttons: [
                {
                  label: 'Public',
                  variant: 'success',
                  onClick: () => handleVisibilitySetting('public'),
                },
                {
                  label: 'Friends Only',
                  variant: 'warning',
                  onClick: () => handleVisibilitySetting('friends'),
                },
                {
                  label: 'Private',
                  variant: 'error',
                  onClick: () => handleVisibilitySetting('private'),
                },
              ],
            });
          },
        },
        {
          label: 'Data Download',
          variant: 'info',
          onClick: () => handleDataDownload(),
        },
        {
          label: 'Activity Settings',
          variant: 'default',
          onClick: () => {
            addMessage({
              type: 'other',
              content: '📈 **Activity Settings**\n\nConfigure what data we collect:',
              buttons: [
                {
                  label: 'Allow Analytics',
                  variant: 'success',
                  onClick: () => {
                    addMessage({
                      type: 'other',
                      content: '✅ Analytics enabled. This helps us improve our service.',
                    });
                  },
                },
                {
                  label: 'Disable Analytics',
                  variant: 'error',
                  onClick: () => {
                    addMessage({
                      type: 'other',
                      content: '❌ Analytics disabled.',
                    });
                  },
                },
              ],
            });
          },
        },
        {
          label: 'Back to Menu',
          variant: 'dull',
          onClick: () => handleBackToMenu(),
        },
      ],
    });
  };

  const handleVisibilitySetting = (setting: string) => {
    addMessage({
      type: 'other',
      content: `✅ Profile visibility set to "${setting}".`,
      buttons: [
        {
          label: 'Main Menu',
          onClick: () => handleBackToMenu(),
        },
      ],
    });
  };

  const handleDataDownload = () => {
    addMessage({
      type: 'other',
      content: '📥 **Download Your Data**\n\nWe\'ll prepare a copy of all your data. This may take a few minutes.',
      buttons: [
        createRequestConfirmationButton({
          initialLabel: 'Request Download',
          confirmationMessage: 'We\'ll email you a download link when your data is ready (usually within 24 hours). Continue?',
          variant: 'info',
          onConfirm: async () => {
            addMessage({
              type: 'other',
              content: 'Preparing your data export...',
            });

            await new Promise(resolve => setTimeout(resolve, 2000));

            addMessage({
              type: 'other',
              content: '✅ Data export requested!\n\nYou\'ll receive an email at your registered email address with a download link within 24 hours.',
              buttons: [
                {
                  label: 'Main Menu',
                  onClick: () => handleBackToMenu(),
                },
              ],
            });
          },
          onReject: () => {
            addMessage({
              type: 'other',
              content: 'Data export cancelled.',
            });
          },
        }),
        {
          label: 'Back',
          variant: 'dull',
          onClick: () => handlePrivacy(),
        },
      ],
    });
  };

  const handleDeleteAccount = () => {
    addMessage({
      type: 'other',
      content: '⚠️ **Delete Account**\n\nWe\'re sorry to see you go. Before you delete your account, please note:\n\n• This action cannot be undone\n• All your data will be permanently deleted\n• You\'ll lose access to all services\n• Active subscriptions will be cancelled\n\nAre you sure you want to continue?',
      buttons: [
        createRequestConfirmationButton({
          initialLabel: 'Yes, Delete My Account',
          confirmationMessage: 'This is your last chance. Are you absolutely sure you want to permanently delete your account and all associated data?',
          variant: 'error',
          onConfirm: () => {
            addMessage({
              type: 'other',
              content: 'Before we proceed, please type "DELETE" to confirm:',
              userResponseCallback: () => {
                const { getMessages } = useChatStore.getState();
                const messages = getMessages();
                const lastSelfMessage = [...messages]
                  .reverse()
                  .find(msg => msg.type === 'self');

                if (lastSelfMessage?.content === 'DELETE') {
                  handleFinalDelete();
                } else {
                  addMessage({
                    type: 'other',
                    content: '❌ Confirmation failed. Please type "DELETE" exactly as shown.',
                    userResponseCallback: () => {
                      // Recurse to check again
                      const messages2 = useChatStore.getState().getMessages();
                      const lastSelfMessage2 = [...messages2]
                        .reverse()
                        .find(msg => msg.type === 'self');
                      
                      if (lastSelfMessage2?.content === 'DELETE') {
                        handleFinalDelete();
                      } else {
                        addMessage({
                          type: 'other',
                          content: '❌ Account deletion cancelled due to incorrect confirmation.',
                        });
                      }
                    },
                  });
                }
              },
            });
          },
          onReject: () => {
            addMessage({
              type: 'other',
              content: '✅ Account deletion cancelled. Your account is safe!',
            });
          },
        }),
        {
          label: 'No, Keep My Account',
          variant: 'success',
          onClick: () => {
            addMessage({
              type: 'other',
              content: '✅ Great! Your account will remain active.',
            });
          },
        },
      ],
    });
  };

  const handleFinalDelete = async () => {
    addMessage({
      type: 'other',
      content: 'Deleting your account...',
    });

    await new Promise(resolve => setTimeout(resolve, 2000));

    addMessage({
      type: 'other',
      content: '👋 Your account has been permanently deleted.\n\nThank you for using our service. We hope to see you again in the future!',
    });
  };

  const handleBackToMenu = () => {
    addMessage({
      type: 'other',
      content: '⚙️ **Account Settings**\n\nWhat would you like to manage?',
      buttons: [
        {
          label: 'Profile Settings',
          variant: 'info',
          onClick: () => handleProfileSettings(),
        },
        {
          label: 'Security',
          variant: 'warning',
          onClick: () => handleSecurity(),
        },
        {
          label: 'Privacy',
          variant: 'default',
          onClick: () => handlePrivacy(),
        },
        {
          label: 'Delete Account',
          variant: 'error',
          onClick: () => handleDeleteAccount(),
        },
      ],
    });
  };

  return (
    <div className="min-h-screen bg-background">
      <Chat initialMessages={initialMessages} theme="dark" />
    </div>
  );
}

export default AccountManagementChat;
```

## Key Features

### 1. Profile Management

- Name updates
- Email changes with verification
- Phone number updates
- Input validation for all fields

### 2. Security Controls

- Password changes with strong validation
- Two-factor authentication setup
- Login history viewing
- Session revocation

### 3. Privacy Settings

- Profile visibility controls
- Data download requests
- Analytics preferences

### 4. Account Deletion

Multi-level confirmation:
- Warning about consequences
- Confirmation dialog
- Typed confirmation ("DELETE")
- Final deletion
